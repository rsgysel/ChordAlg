cmake_minimum_required(VERSION 2.8 FATAL_ERROR)

function(add_chordalg_executable arg1)
    add_executable(${arg1} ${arg1}.cxx)
    target_link_libraries(${arg1} ChordAlgLib)
endfunction()

add_chordalg_executable(Atoms)
add_chordalg_executable(AtomsExperiment)
add_chordalg_executable(AtomSeparatorGraphExperiment)
add_chordalg_executable(CellIntersectionGraph)
add_chordalg_executable(Composite_heuristic)
add_chordalg_executable(MCS_triangulation)
add_chordalg_executable(Mixed_heuristic)
add_chordalg_executable(MMC_heuristic)
add_chordalg_executable(MMC_MaxCompatSupertree)
add_chordalg_executable(PDRS_ratio_heuristic)
add_chordalg_executable(PDRS_wsum_heuristic)
add_chordalg_executable(SeparatorGraphExperiment)
